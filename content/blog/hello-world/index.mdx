# momo

```javascript
export const fightersReducer = (state, action) => {
  switch (action.type) {
    case fightersActionTypes.FETCHING:
      return {
        fetchState: REQUEST_STATE.LOADING,
      }
    case fightersActionTypes.FETCH_SUCCESS:
      return {
        fetchState: REQUEST_STATE.OK,
      }
    default:
      throw new Error()
  }
}
```
